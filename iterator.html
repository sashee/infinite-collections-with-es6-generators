An infinite collection using iterators:

			var naturalNums = {
				[Symbol.iterator]: (()=>{
					let i = 0;
					return {
						next: () => {
							return {done: false, value: i++};
						}
					}
				})
			};

			for(let i of naturalNums){
				if(i > 10) break; // Don't forget to terminate!
				document.write(i+"<br/>");
			}
